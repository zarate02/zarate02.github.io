# to JsonPath
- k get node -o=json > /opt/outputs/nodes.json
- k get node -o=jsonpath='{.items[*].metadata.name}' > /opt/outputs/node_names.txt
- k get pv --sort-by=spec.capacity.storage -o custom-columns='NAME:.metadata.name,CAPACITY:spec.capacity.storage'
- kubectl config view --kubeconfig my-kube-config  -o jsonpath='{.contexts[?(@.context.user=="aws-user")].name}'
